6 Assign a function to a variable

To cal_bmi variable = set function name as cal_bmi

cal_bmi<-function(height,weight){
    if(height>5){
        height <- height/100
    }
    bmi <- weight / height**2
    return(bmi)
}
cal_bmi(160,50)
## [1] 20
cal_bmi(1.6,50)
## [1] 20
